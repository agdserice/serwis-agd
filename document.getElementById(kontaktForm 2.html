document.getElementById("kontaktForm").addEventListener("submit", function(event) { event.preventDefault(); let name = document.getElementById("name").value; let email = document.getElementById("email").value; let message = document.getElementById("message").value; if (name && email && message) { document.getElementById("status").innerText = "Wiadomość wysłana!"; document.getElementById("status").style.color = "green"; this.reset(); } else { document.getElementById("status").innerText = "Wypełnij wszystkie pola!"; document.getElementById("status").style.color = "red"; } });
